def bash_like_comparsion(sign, value1, value2)
  case sign
  when :eq
    value1 == value2
  when :gt
    value1 > value2
  when :gteq
    value1 >= value2
  when :lt
    value1 < value2
  when :lteq
    value1 <= value2
  end
end
